快速幂(A ^ B mod C) ​

  • 原理是利用了所有整数都是能够转换为二进制的形式。
  • 将 B 转换为二进制，方法：P = 2 * (P/2) + P%2

则 B = 19 = 10011(自底向上) = 2^4 + 2^1 + 2^0

得到 A 2^4 + 2^1 + 2^0 = (A 2^4 *A 2^1 *A 2^0 )mod C

= (A 2^4 mod C )X(A 2^1 mod C )X(A 2^0 mod C)mod C

C++
#define int long long
int fpow(int num1, int num2, int modd)
{
    int ans = 1 % modd;
    while(num2)
    {
        if(num2&1)
            ans = (ans * num1) % modd;// 遇到1乘入，遇到0就跳过
        num1 = (num1 * num1) % modd;
        num2 >>= 1;
    }
    return ans;
} // 题目中出现 A^BmodC --> 快速幂

最大子段和 ​

c++
#define int long long
const int MAX = 2e5 + 10;
int a[MAX] = {0};
const int inf = 1e17 + 10; // long long 下的 近似于 正无穷
int solve(int l, int r)
{
    if (l > r)
        return -inf;
    int mid = (l + r) >> 1;
    int ans1 = solve(l, mid - 1); // 在mid左边的子段中找到最大子段和
    int ans2 = solve(mid + 1, r); // 同理
    int ans3, sumL = -inf, sumR = -inf;
    for (int i = mid, sum = 0; i >= l; --i)
    {
        sum += a[i]; // 子段，一直+a[i]
        sumL = max(sumL, sum);
    }
    for (int j = mid, sum = 0; j <= r; ++j)
    {
        sum += a[j];
        sumR = max(sumR, sum);
    }
    ans3 = sumL + sumR - a[mid]; // 都是从mid开始的前后缀和，所以 - a[mid]
    return max({ans1, ans2, ans3});//求出三段的最大值
}

归并排序(求逆序对) ​

逆序对:
在升序的数组中发现 0 < i < j < n and a[i] > a[j] 说明 a[i] a[j] 为一个逆序对。又因为左边(l,mid)数组已经是有序的，所以当下标 i 不为 mid 的数大于 j 时，从 i 到 mid 的数都要大于 j 下标。 求最小交换次数： 因为每交换一次就会减少一个逆序对，所以求最小交换次数就是求所有的逆序对。

c++
const int MAX = 1e5 + 10;
vector<int> a(MAX);
void merge(int mid, int l, int r)
{
    static vector<int> w;
    w.clear();//清空size
    int i = l, j = mid + 1;
    while (i <= mid && j <= r)
    {
        if (a[i] <= a[j])
            w.push_back(a[i++]);
        else
        {
            w.push_back(a[j++]);
            // cnt += mid - i + 1; // 记录逆序对个数
        }
    }
    while (i <= mid)
        w.push_back(a[i++]);
    while (j <= r)
        w.push_back(a[j++]);
    for (int i = 0, j = l; i < w.size(); ++i, ++j) // 从left开始，传进来的是l，r
        a[j] = w[i];
    //然后所有的值都排好序放到w数组中，重排了l到r之间元素
}
void merge_sort(int l, int r)
{
    if (l >= r)
        return;
    int mid = l + ((r - l) / 2);
    merge_sort(l, mid);//bug?原因是因为当size = 5时，mid = 2，所以如果是，merge_sort(l, mid-1),则(0,1)(3,4)导致下标为2的元素不被考虑到。
    merge_sort(mid + 1, r);
    merge(mid, l, r);
}

快速排序 ​

c++
void quick(vi& arr, int begin, int end) {
    // 递归，直到 start = end 为止
    if (begin > end) return;
    int base = arr[begin], i = begin, j = end;
    while (i != j) {
        // 从右向左找比基准数小的数 （要先从右边开始找）
        while (arr[j] >= base && i < j) j--;
        // 从左向右找比基准数大的数
        while (arr[i] <= base && i < j) i++;
        // 主要是找到目标数的下标，然后再交换位置
        if (i < j) swap(arr[i], arr[j]);
    }
    // 最终将基准数归位
    arr[begin] = arr[i];
    arr[i] = base;
    // 第一趟把基准数放到中间后，分左右两边依此快排
    quick(arr, begin, i - 1); // 继续处理左边的
    quick(arr, i + 1, end); // 继续处理右边的
}

求区间中小于 m 的最大和

C++
void solve()// 看到序列不保证有序，但都大于0，想到了前缀和 + 二分
{
    n = read(), m = read();
    for (int i = 1; i <= n; ++i)
    {
        num[i] = read();
        num[i] += num[i - 1]; // 将原数组直接转化为前缀和
    }
    int px, py, ans = -1e6;
    for (int i = 1; i <= n; ++i) // 设i 为该区间的左端点
    {
        int l = i - 1, r = n;  // l r 为寻找区间右端点的左右范围
        while (l <= r) // 跳出条件为 l = r + 1; 且答案为r
        {
            int mid = (l + r) / 2;
            if (num[mid] - num[i - 1] <= m)// 因为前缀和非递减，并且要找到区间最大值 则左半边必然无法得到更大的。
            {
                l = mid + 1;
            }
            else // mid 取值过大，需要将右半边删除
                r = mid - 1;
        }
        if (num[r] - num[i - 1] <= m && num[r] - num[i - 1] > ans) // 判断 r是否找到，并且求出所有有效区间中最大；条件为输出i小的，只要i从小遍历，必然先找到i小
        {
            px = i;
            py = r;
            ans = num[r] - num[i - 1];
        }
    }
    cout << px << " " << py << " " << ans << endl;
}

cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int MAX = 1e5 + 100;
int n, m;
double sum = 0;
vector<int> num(MAX);
inline int read()
{
    int ans = 0, f = 1;
    char ch = getchar();
    while (!isdigit(ch))
    {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (isdigit(ch))
    {
        ans = (ans << 1) + (ans << 3) + (ch ^ 48);
        ch = getchar();
    }
    return ans * f;
}
bool check(double mid)
{
    int ans = 0;
    for (int i = 1; i <= n; ++i)
    {
        int tmp = num[i] / mid;// double类型赋值给int,自动向下取整
        // 拓展：四舍五入为 (int)(num1/num2 + 0.5)
        ans += tmp;
        if (ans >= m)
            return true;
    }
    return false;
}
void solve()
{
    n = read(), m = read();
    for (int i = 1; i <= n; ++i)
        num[i] = read(), sum += num[i];
    double l = 1, r = sum / m, ans = -1;
    for (int i = 0; i <= 200; i++) //
    {
        double mid = ((r - l) >> 1) + l;
        if (check(mid))
        {
            ans = mid;
            l = mid;
        }
        else
            r = mid;
    }
    if (ans == -1)
        cout << "0" << endl;
    else
        cout << fixed << setprecision(2) << ans << endl;
}
signed main()
{
    solve();
    return 0;
}

难点为条件特判！

题解 P2678 【跳石头】 - ShawnZhou 的博客 - 洛谷博客 (luogu.com.cn)

C++
#include <bits/stdc++.h>
using namespace std;
const int MAX = 5e4 + 100;
int l, n, m, star = 0;
vector<int> arr(MAX);
bool check(int x)
{
    int tmp = 0, cet = 0;
    for (int i = 0; i <= n; ++i)
    {
        if (arr[i] - tmp < x) // 小了就不行，要求最               大值
        {
            cet++; // 删除了，所以就不用将tmp赋值，因为已经不thisNumberExists了，下次还是和原tmp比较。
        }
        else
            tmp = arr[i];
    }
    return cet <= m;
}
int main()
{
    ios::sync_with_stdio(false), cin.tie(nullptr);
    cin >> l >> n >> m;
    for (int i = 0; i < n; ++i)
    {
        cin >> arr[i];
    }
    arr[n] = l;
    int left = 1, right = l;
    while (left <= right)
    {
        int mid = ((right - left) >> 1) + left;
        // 之所以能够log级别，是因为每次都排除了一半的可能的数。
        // 即如果在mid的条件下删除都不可能成功，那那些大于mid就更不用考虑了，right = mid - 1。
        // 但又是要求最短中的最大。这里最短是相对于整个全局的最短。
        if (check(mid))
        {
            left = mid + 1;
        }
        else
            right = mid - 1;
    }
    cout << left - 1;
    return 0;
}

#define eps 1e-8
double a, b, c, d;
double check(double num)
{
    return a * num * num * num + b * num * num + c * num + d;
}
int main()
{
    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    cin >> a >> b >> c >> d;
    int cnt = 0;
    for (int i = -100; i < 100; ++i)//规定一个为1的区间内最多只有一个实根。
    {

        double l = i, r = i + 1; // 在每个[i, i + 1]的区间中去找，
        double num1 = check(l), num2 = check(r);
        if (num1 == 0)
        {
            cout << fixed << setprecision(2) << l << " ";
            cnt++;
            continue;
        }
        if (num1 * num2 < 0) // 若内部有值，则肯定能够通过二分找到
        {
            while (r - l > eps)
            {
                double mid = (r + l) / 2;
                if (check(mid) * check(r) <= 0)
                    l = mid;
                else
                    r = mid;
            }
            if (abs(check(r)) <= 1e-4) // 0.0001，多余的判断...
            {
                cout << fixed << setprecision(2) << r << " ";
                cnt++;
            } // 可直接用 printf("%.2f\n",r);
        }
        if (cnt == 3)
            break;
    }
    return 0;
}

  • 考点为将题目中的公式转换成 sum >= 0
  • 即 $\sum_{i=1}^n$num2[i] / $\sum_{i=1}^n$num1[i] >= maxx -->
  • --> (num2[1] - num1[1] _ maxx) + (num2[2] - num1[2] _ maxx) + ... = 0

    while (!str1.empty() && !str2.empty())
    {
        flag = (str1.back() > str2.back());
        if (!flag && num1 == k)
            flag = 1;
        else if (flag && num2 == k)
            flag = 0;
        if (flag)

            str3.push_back(str1.back()), num1++, num2 = 0, str1.pop_back();

        tmp1 = tmp2, tmp2 = ans;

    return ans;

{
    if (num == 1)//基本情况处理
    {
        Movee(source, target);
        return;

    Hanoi(num - 1, source, target, temp);//递归调用，超级操作
    Movee(source, target);//微操作
    Hanoi(num - 1, temp, source, target);//递归调用，超级操作
}
