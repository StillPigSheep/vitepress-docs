Miller_Robin 素数检测 ​

c++
#define ll long long
#define ld long double
#define int long long
#define ull unsigned long long
int qmul(ll a,ll b,ll mod)//快速乘
{
    ll c=(ld)a/mod*b;
    ll res=(ull)a*b-(ull)c*mod;
    return (res+mod)%mod;
}
int qpow(ll a,ll n,ll mod)//快速幂
{
    ll res=1;
    while(n)
    {
        if(n&1) res=qmul(res,a,mod);
        a=qmul(a,a,mod);
        n>>=1;
    }
    return res;
}
bool MRtest(ll n)//Miller Rabin Test
{
    if(n<3||n%2==0) return n==2;//特判
    ll u=n-1,t=0;
    while(u%2==0) u/=2,++t;
    ll ud[]={2,325,9375,28178,450775,9780504,1795265022};
    for(ll a:ud)
    {
        ll v=qpow(a,u,n);
        if(v==1||v==n-1||v==0) continue;
        for(int j=1;j<=t;j++)
        {
            v=qmul(v,v,n);
            if(v==n-1&&j!=t){v=1;break;}//出现一个n-1，后面都是1，直接跳出
            if(v==1) return 0;//这里代表前面没有出现n-1这个解，二次检验失败
        }
        if(v!=1) return 0;//Fermat检验
    }
    return 1;
}

欧拉筛法 ​

  • if(i % prime[j] == 0) break;区别于埃氏筛的原因，导致了其的复杂度为 O(n)(每个数只被筛过一次) 保证都是被最小质因数整除
  • 解释：若 i % prime[j] 得出 i = prime[j] _ a, 且 1 < a < i, 而对于 prime[j]大的素数 prime[j + 1], i _ prime[j + 1] == prime[j] _ a _ prime[j + 1]; 后面的话， prime[j + 1] * i 已经被 prime[j]筛掉了，所以就得提前 break，防止重复。
    c++
    vector<int>prime;
vector<bool>is_prime(MAX2, true);

{
    is_prime[1] = false, is_prime[0] = false;
    for(int i = 2; i <= n; ++i)
    {
        if(is_prime[i])
            prime.push_back(i);
        for(int j = 0; j < prime.size() && i * prime[j] <= n; ++j)
        {
            is_prime[i * prime[j]] = false;
            if(i % prime[j] == 0) break;
        }
    }
}

最大公约数 ​

  1. gcd(a,b) = gcd(a, a+b) = gcd(a, ka+b)
  2. gcd(ka, kb) = k·gcd(a, b)
  3. 定义多个整数的最大公约数：gcd(a, b, c) = gcd(gcd(a, b), c)
  4. 若 gcd(a, b) = d，则 gcd(a/d, b/d) = 1，即 a/d 与 b/d 互素。这个定理很重要。
  5. gcd(a+cb, b) = gcd(a, b)

    原因：更相减损法的 a -= b, 直到了 a == b 为最大公约数

归并排序(求逆序对) ​

逆序对: 在升序的数组中发现 0 < i < j < n and a[i] > a[j] 说明 a[i] a[j] 为一个逆序对。又因为左边(l,mid)数组已经是有序的，所以当下标 i 不为 mid 的数大于 j 时，从 i 到 mid 的数都要大于 j 下标。 求最小交换次数： 因为每交换一次就会减少一个逆序对，所以求最小交换次数就是求所有的逆序对。

c++
const int MAX = 1e5 + 10;
vector<int> a(MAX);
void merge(int mid, int l, int r)
{
    static vector<int> w;
    w.clear();//清空size
    int i = l, j = mid + 1;
    while (i <= mid && j <= r)
    {
        if (a[i] <= a[j])
            w.push_back(a[i++]);
        else
        {
            w.push_back(a[j++]);
            // cnt += mid - i + 1; // 记录逆序对个数
        }
    }
    while (i <= mid)
        w.push_back(a[i++]);
    while (j <= r)
        w.push_back(a[j++]);
    for (int i = 0, j = l; i < w.size(); ++i, ++j) // 从left开始，传进来的是l，r
        a[j] = w[i];
    //然后所有的值都排好序放到w数组中，重排了l到r之间元素
}
void merge_sort(int l, int r)
{
    if (l >= r)
        return;
    int mid = l + ((r - l) / 2);
    merge_sort(l, mid);//bug?原因是因为当size = 5时，mid = 2，所以如果是，merge_sort(l, mid-1),则(0,1)(3,4)导致下标为2的元素不被考虑到。
    merge_sort(mid + 1, r);
    merge(mid, l, r);
}

c++
int find(int x) //在查找中要将顺带的递归记录到根节点权值。
{
    if (x == fa[x])
        return x;
    int t = fa[x];
    fa[x] = find(fa[x]);
    // 加上原父节点，此时已经是递归完后的父节点，sum[t]已经记录了到根节点
    sum[x] += sum[t];
    return fa[x];
}
void merge(int x, int y, int z)
{
    int ans1 = find(x), ans2 = find(y);
    if (ans1 == ans2) //若两个根节点相同后，表示sum[x] ==  x--> y -->...--> root 和 sum[y] == y-->...-->root z表示x --> y则等于sum[x] - sum[y]
    {
        if (sum[x] - sum[y] != z)
        {
            cnt++;
        }
    }
    else
    {
        fa[ans1] = ans2;
        sum[ans1] = sum[y] - sum[x] + z; //公式推导
    }
}

// chain of food 扩展域
#include <iostream>
#include<cstdio>
using namespace std;
const int MAX = 5e4 + 100;
int fa[MAX * 3] = {0};
int n, k;
int cnt = 0;
void init()
{
    for (int i = 1; i <= n * 3; ++i)
    {
        fa[i] = i;
    }
}
int find(int x)
{
    if (fa[x] == x)
        return x;
    return fa[x] = find(fa[x]);
}
void merge(int x, int y)
{
    fa[find(x)] = find(y);
}
int main()
{
    cin >> n >> k;
    init();
    for (int i = 0, tmp, c1, c2; i < k; ++i)
    {
        scanf("%d %d %d", &tmp, &c1, &c2);
        if (c1 > n || c2 > n)
        {
            cnt++;
            continue;
        }
        if (tmp == 1)
        {
            //从图中可以看出，当c1, c2为同类时，c1 只可能等于 c2 ,所以需要排除两种情况，c1 == c2 + n, c1 == c2 + n * 2
            if (find(c1) == find(c2 + n) || find(c1) == find(c2 + 2 * n))
            {
                cnt++;
            }
            else
            {
                merge(c1, c2);
                merge(c1 + n, c2 + n);
                merge(c1 + 2 * n, c2 + 2 * n);
            }
        }
        else
        {
            // 因为为c1->c2 c1吃c2,和c1为同类的是c1 == c2 + 2 * n 所以要排除，c1 == c2 + n c1 == c2
            if (c1 == c2 || find(c1) == find(c2 + n) || find(c1) == find(c2))
            {
                cnt++;
            }
            else
            {
                merge(c1, c2 + 2 * n);
                merge(c1 + n, c2);
                merge(c1 + 2 * n, c2 + n);
            }
        }
    }
    cout << cnt << endl;
    return 0;
}

单调队列 ​

经典例题：滑动窗口 “单调队列”即单调:单调递增或递减; 队列:说明有出队入队时间复杂度为 O(1); 可以使用 STL 的 deque 维护最大值或最小值。

步骤

  1. 判断是否需要将队头出队，原因是滑动窗口，当队头不在窗口内，则出队。
  2. 队尾入队操作，如果求最大值，队尾的值小于等于新入队的值，则需要将队尾元素从队尾出队(原因是新进入的元素都比它前面的大，那当前窗口是新加入的最大，则之前元素不需要被考虑了)，一直出队，直到有元素是大于新入队元素，则新入队元素入队。
  3. 输出当前的队头表示维护的最大值。
c++
#include<c++>
using namespace std;
#define int long long
const int MAX = 1e6 + 10;
deque<int>que;
vector<int>arr(MAX);
signed main()
{
    int n, m;
    cin >> n >> m;
    for(int i = 0; i < n; ++i)
        cin >> arr[i];
    // min
    for(int i = 0; i < n; ++i)
    {
        if(!que.empty() && que.front() < i - m + 1)
            que.pop_front();
        while(!que.empty() && arr[que.back()] >= arr[i])//求最小值，找到一个小于新来元素
            que.pop_back();
        que.push_back(i);
        if(i >= m - 1) cout << arr[que.front()] << " ";
    }
    cout << endl;
    // max
    que.clear();
    for(int i = 0; i < n; ++i)
    {
        if(!que.empty() && que.front() < i - m + 1)
            que.pop_front();
        while(!que.empty() && arr[que.back()] <= arr[i]) ///求最大值，找到一个大于新来元素
            que.pop_back();
        que.push_back(i);
        if(i >= m - 1) cout << arr[que.front()] << " ";
    }
    return 0;
}

        cir.pop();            //出圈

            }
        }
    }

注意：用 set 或 map 时，结构体 node 的比较函数只按照 x 的大小进行比较，而没有考虑 y 的大小。如果两个 node 的 x 相同，但 y 不同，它们在 set 中被视为相等，可能会导致找不到某些元素。需要 x,y 都考虑。

c++
struct node{
    int x, y;
    bool operator<(const &node nextt) const{
        if(x != nextt.x)
            return x < nextt.x;
        return y < nextt.y;
    }
}

使用 stl 中的 next_permutation 来求全排列数 ​

  1. 用于求一个集合中的某些个数的值

    可知，使用了 next_permutation 是不对数组自身进行操作的，所以我们可以将数组最后一位的值改为 1，然后逐渐地递减到 0，以达到所有的 0 都变成了 1，实现 00...--->111...的全排列。

c++
int n, cet = 1;
cin >> n;
int flag[10] = {0};
for (int i = 0; i < n; ++i)
{
    flag[n - cet] = 1;
    do
    {
        for (int i = 0; i < n; ++i)
        {
            cout << flag[i] << " ";
        }
        cout << endl;
    } while (next_permutation(flag, flag + n));
    cet++;
}

一般地，在我们比赛/刷题过程中，当输入数字数量(次数)超过 $ 5 \times 10^{4} $时使用它

c++
inline int read()
{
    int x = 0, f = 1;
    char ch = getchar();
    while(ch < '0'|| ch > '9') // 防止 cin 和 read 混用后出问题
    {
        if(ch == '-')
            f = -1;
        ch = getchar();
    }
    while(ch >= '0' && ch <= '9')
    {
        x = (x << 1) + (x << 3) + (ch ^ 48);
        //等价于 x = (x + (x << 2))<< 1 + (ch ^ 48);
        ch = getchar();
    }
    return x * f;
}
