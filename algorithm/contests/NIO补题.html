#include<bits/stdc++.h>
using namespace std;
const int MAX = 2e6 + 10;
vector<string>str(MAX);
void merge(int mid, int l, int r)
{
    static vector<string>w;
    w.clear();
    int i = l, j = mid + 1;
    while(i <= mid && j <= r)
    {
        if(str[i] + str[j] >= str[j] + str[i])
            w.push_back(str[j++]);
        else 
            w.push_back(str[i++]);
    }
    while(i <= mid) w.push_back(str[i++]);
    while(j <= r) w.push_back(str[j++]);
    for(int i = 0; i < w.size(); ++i)
        str[l + i] = w[i]; 
}
void merge_sort(vector<string> & q,int l, int r)
{
    if (l >= r)
        return;
    int mid = (l + r) / 2;
    merge_sort(q,l,mid);
    merge_sort(q,mid+1,r);
    merge(mid, l, r);
}
int main()
{
    int n;
    cin >> n;
    for(int i = 0; i < n; ++i)
        cin  >> str[i];
    merge_sort(str,0,n - 1);
    for(int i = 0; i < n; ++i)
        cout << str[i];
    return 0;
}
